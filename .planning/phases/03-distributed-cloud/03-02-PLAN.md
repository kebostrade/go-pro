# Research Recommendations (03-RESEARCH.md)
- Use JetStream for all new event-driven work (persistence, delivery guarantees)
- Don't use core NATS for new event-driven work
- Always Ack messages (messages won't be redelivered without acknowledgment)
- Use queue groups for distributed worker pools

# JetStream Pattern
```go
// Connect with JetStream
nc, _ := nats.Connect(nats.DefaultURL)
js, _ := jetstream.New(nc)

// Publish with persistence
js.Publish(ctx, "events.orders", data)

// Subscribe with consumer
sub, _ := js.Subscribe(ctx, "events.orders", func(msg jetstream.Msg) {
    msg.Ack()  // Acknowledge
})
```

       ```go
       type Publisher struct {
           js jetstream.JetStream
       }

       func NewPublisher(nc *nats.Conn) (*Publisher, error) {
           js, err := jetstream.New(nc)
           if err != nil {
               return nil, err
           }
           return &Publisher{js: js}, nil
       }

       func (p *Publisher) Publish(ctx context.Context, subject string, data []byte) error {
           _, err := p.js.Publish(ctx, subject, data)
           return err
       }

       func (p *Publisher) PublishJSON(ctx context.Context, subject string, v interface{}) error {
           data, err := json.Marshal(v)
           if err != nil {
               return err
           }
           _, err = p.js.Publish(ctx, subject, data)
           return err
       }
       ```

    2. **internal/jetstream/subscriber.go** - JetStream subscriber with ack:
       ```go
       package jetstream

       type Subscriber struct {
           js jetstream.JetStream
       }

       type MessageHandler func(msg jetstream.Msg)

       func NewSubscriber(nc *nats.Conn) (*Subscriber, error) {
           js, err := jetstream.New(nc)
           if err != nil {
               return nil, err
           }
           return &Subscriber{js: js}, nil
       }

       func (s *Subscriber) Subscribe(ctx context.Context, subject string, handler MessageHandler) error {
           _, err := s.js.Subscribe(ctx, subject, func(msg jetstream.Msg) {
               handler(msg)
               msg.Ack()  // Acknowledge after processing
           })
           return err
       }
       ```

       ```go
       package queue

       type Worker struct {
           nc       *nats.Conn
           group    string
           subject  string
           handler  func([]byte) error
       }

       func NewWorker(nc *nats.Conn, group, subject string, handler func([]byte) error) *Worker {
           return &Worker{
               nc:       nc,
               group:    group,
               subject:  subject,
               handler:  handler,
           }
       }

       func (w *Worker) Start(ctx context.Context) error {
           _, err := w.nc.QueueSubscribe(w.subject, w.group, func(msg *nats.Msg) {
               if err := w.handler(msg.Data); err != nil {
                   // NAK and requeue on failure
                   msg.Nak()
               } else {
                   msg.Ack()
               }
           })
           return err
       }
       ```
