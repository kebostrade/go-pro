    Add to frontend/src/lib/api.ts:

    ```typescript
    interface ReviewSubmission {
      id: string;
      user_id: string;
      topic_id: string;
      exercise_id: string;
      code: string;
      feedback: string;
      submitted_at: string;
    }

    interface SubmitReviewRequest {
      user_id: string;
      topic_id: string;
      exercise_id: string;
      code: string;
    }

    // Submit code for AI review
    async function submitReview(request: SubmitReviewRequest): Promise<ReviewSubmission> {
      const response = await fetch('/api/review/submit', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(request),
      });
      if (!response.ok) {
        throw new Error(`Review failed: ${response.statusText}`);
      }
      return response.json();
    }

    // Get user's review history
    async function getReviewHistory(userId: string): Promise<ReviewSubmission[]> {
      const response = await fetch(`/api/review/history?user_id=${userId}`);
      if (!response.ok) {
        throw new Error(`Failed to fetch history: ${response.statusText}`);
      }
      return response.json();
    }
    ```
