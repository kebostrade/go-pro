    ```go
    func (h *ReviewHandler) SubmitReview(w http.ResponseWriter, r *http.Request) {
        var req SubmitReviewRequest
        if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
            http.Error(w, "Invalid request", http.StatusBadRequest)
            return
        }

        // Get user's code from request
        code := req.Code

        // Call AI agent for analysis
        feedback, err := h.analyzeCode(r.Context(), code)
        if err != nil {
            http.Error(w, "Analysis failed: "+err.Error(), http.StatusInternalServerError)
            return
        }

        // Store submission
        submission := &ReviewResponse{
            ID:          generateID(),
            UserID:      req.UserID,
            TopicID:     req.TopicID,
            ExerciseID:  req.ExerciseID,
            Code:        code,
            Feedback:    feedback,
            SubmittedAt: time.Now(),
        }

        // TODO: Store in database

        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(submission)
    }
    ```

    4. analyzeCode helper:
    ```go
    func (h *ReviewHandler) analyzeCode(ctx context.Context, code string) (string, error) {
        // Use CodeAnalysisTool to analyze code
        result, err := h.codeAnalysisTool.Analyze(ctx, code)
        if err != nil {
            return "", err
        }

        // Format as conversational feedback
        return formatConversationalFeedback(result), nil
    }

    func formatConversationalFeedback(result *types.AnalysisResult) string {
        var feedback strings.Builder

        feedback.WriteString("Great job submitting your code! ")

        if len(result.Issues) == 0 {
            feedback.WriteString("Your code looks clean and follows good practices. ")
            feedback.WriteString("Consider adding some comments to explain your logic.")
            return feedback.String()
        }

        feedback.WriteString("I reviewed your code and have a few suggestions:\n\n")

        // Group by category
        if len(result.SecurityIssues) > 0 {
            feedback.WriteString("**Security:**\n")
            for _, issue := range result.SecurityIssues {
                feedback.WriteString("- " + issue.Description + "\n")
            }
            feedback.WriteString("\n")
        }

        if len(result.PerformanceIssues) > 0 {
            feedback.WriteString("**Performance:**\n")
            for _, issue := range result.PerformanceIssues {
                feedback.WriteString("- " + issue.Description + "\n")
            }
            feedback.WriteString("\n")
        }

        if len(result.BestPractices) > 0 {
            feedback.WriteString("**Best Practices:**\n")
            for _, issue := range result.BestPractices {
                feedback.WriteString("- " + issue.Description + "\n")
            }
        }

        feedback.WriteString("\nOverall score: " + fmt.Sprintf("%d/100", result.OverallScore))

        return feedback.String()
    }
    ```
